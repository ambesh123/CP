
int ld[MAXN];

void m_sieve(){
    fr(i,MAXN)ld[i]=i;

    for(int i=2;i*i<MAXN;i++){
        if(ld[i]==i){
            for(int j=2*i;j<MAXN;j+=i){
                if(ld[j]==j){ld[j]=i;}
            }
        }
    }
}

vector<int> factors(int x){
    vector<int> ret;
    int tmp = -1;
    while(x > 1){
        tmp = ld[x];
        ret.push_back(tmp);
        while(ld[x] == tmp) x /= ld[x];
    }
    return ret;
}
