void sieve(){

    prime[0]=prime[1]=false;
    for(int i=2;i*i<MAXN;i++){
        for(int j=2*i;j<MAXN;j+=i)prime[j]=false;
    }
    fr(i,MAXN)if(prime[i])primes.pb(i);
}


ll modPower(ll x,ll y){
    x%=mod;
    ll res = 1;
    while(y){
        if(y&1)res= (res*x)%mod;
        x = (x*x)%mod;
        y>>=1;
    }
    return res;
}
